    Function mutate

    • mutate(_value: unknown, _context: ClassPropertyContext): void

      Decorator to include properties of a Mutable in its Mutator (via Mutable.getMutator). Use on getters to include them in the mutator and display them in the editor.

      Usage: Apply this decorator to a property.

      Example:

      export class SomeScript extends ƒ.ComponentScript {
        #size: number = 1;

        @ƒ.mutate // apply the decorator to the getter
        public get size(): number {
          return this.#size;
        }

        // define a setter to allow writing, or omit it to leave the property read-only
        public set size(_size: number) {
          this.#size = _size;
        }
      }
